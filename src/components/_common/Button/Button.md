### Standard Buttons

```tsx
<div style={{ padding: 10 }}>
  <Button size="small">
    small
  </Button>
</div>
<div style={{ padding: 10 }}>
  <Button size="medium">
    medium
  </Button>
</div>
<div style={{ padding: 10 }}>
  <Button size="large">
    large
  </Button>
</div>
```

### Button With Icon
```tsx
<div style={{ padding: 10 }}>
  <Button size="medium" withIcon>
    Select <span>&darr;</span>
  </Button>
</div>
<div style={{ padding: 10 }}>
  <Button size="small" variant="flat" withIcon>
    Edit <span>&#9998;</span>
  </Button>
</div>
```
